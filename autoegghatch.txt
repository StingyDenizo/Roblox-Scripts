local currentegg = nil

function autohatch(eggname)
	if eggname == "nightmare-egg" then
		currentegg = "Nightmare Egg"
	elseif eggname == "rainbow-egg" then
		currentegg = "Rainbow Egg"
	elseif eggname == "void-egg" then
		currentegg = "Void Egg"
	elseif eggname == "event-1" then
		currentegg = "Bunny Egg"
	elseif eggname == "event-2" then
		currentegg = "Pastel Egg"
	elseif eggname == "man-egg" then
		currentegg = "Aura Egg"
	end
end

while true do
	wait()
	if currentegg ~= nil then
		wait(0.3)
		
		local args = {
			"HatchEgg",
			currentegg,
			6
		}
		game:GetService("ReplicatedStorage"):WaitForChild("Shared"):WaitForChild("Framework"):WaitForChild("Network"):WaitForChild("Remote"):WaitForChild("Event"):FireServer(unpack(args))

	end
end
